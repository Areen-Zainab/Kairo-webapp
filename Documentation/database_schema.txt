# Kairo Webapp Database Schema
# Comprehensive database schema matching the exact Prisma definition (backend/prisma/schema.prisma)

## Core Tables

### Users
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    email TEXT UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    name TEXT NOT NULL,
    profile_picture_url TEXT,
    audio_sample_url TEXT,
    timezone TEXT DEFAULT 'UTC',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    last_login TIMESTAMP,
    is_active BOOLEAN DEFAULT TRUE,
    email_verified BOOLEAN DEFAULT FALSE,
    biometric_consent BOOLEAN DEFAULT FALSE,
    consent_given_at TIMESTAMP
);

### User Sessions
CREATE TABLE user_sessions (
    id SERIAL PRIMARY KEY,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    session_token TEXT UNIQUE NOT NULL,
    device_info TEXT,
    ip_address TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_activity TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    expires_at TIMESTAMP NOT NULL,
    is_active BOOLEAN DEFAULT TRUE
);

### User Preferences
CREATE TABLE user_preferences (
    id SERIAL PRIMARY KEY,
    user_id INT UNIQUE NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    auto_join BOOLEAN DEFAULT FALSE,
    auto_record BOOLEAN DEFAULT FALSE,
    default_duration INT DEFAULT 30,
    timezone TEXT DEFAULT 'UTC',
    theme_mode TEXT DEFAULT 'light',
    accent_color TEXT DEFAULT '#007bff',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP
);

### Notification Settings
CREATE TABLE notification_settings (
    id SERIAL PRIMARY KEY,
    user_id INT UNIQUE NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    email_meeting_reminders BOOLEAN DEFAULT TRUE,
    email_meeting_summaries BOOLEAN DEFAULT TRUE,
    email_action_items BOOLEAN DEFAULT TRUE,
    email_weekly_digest BOOLEAN DEFAULT FALSE,
    push_meeting_starting BOOLEAN DEFAULT TRUE,
    push_meeting_joined BOOLEAN DEFAULT FALSE,
    push_mentions_and_replies BOOLEAN DEFAULT TRUE,
    push_action_items_due BOOLEAN DEFAULT TRUE,
    in_app_meeting_updates BOOLEAN DEFAULT TRUE,
    in_app_transcription_ready BOOLEAN DEFAULT TRUE,
    in_app_shared_with_you BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP
);

### Reminder Preferences
CREATE TABLE reminder_preferences (
    id SERIAL PRIMARY KEY,
    user_id INT UNIQUE NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    enabled BOOLEAN DEFAULT TRUE,
    reminder_intervals INTEGER[] DEFAULT '{24, 1}',
    quiet_hours_start INT,
    quiet_hours_end INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP
);

### Workspaces
CREATE TABLE workspaces (
    id SERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    description TEXT,
    code TEXT UNIQUE NOT NULL,
    owner_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    color_theme TEXT DEFAULT '#9333ea',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    is_active BOOLEAN DEFAULT TRUE,
    is_archived BOOLEAN DEFAULT FALSE,
    archived_at TIMESTAMP,
    archived_by INT,
    settings JSONB
);

### Workspace Members
CREATE TABLE workspace_members (
    id SERIAL PRIMARY KEY,
    workspace_id INT NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    role TEXT DEFAULT 'member',
    joined_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    is_active BOOLEAN DEFAULT TRUE,
    permissions JSONB,
    UNIQUE(workspace_id, user_id)
);

### Workspace Invites
CREATE TABLE workspace_invites (
    id SERIAL PRIMARY KEY,
    workspace_id INT NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
    invited_email TEXT NOT NULL,
    invited_user_id INT REFERENCES users(id) ON DELETE CASCADE,
    invited_by INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    role TEXT DEFAULT 'member',
    status TEXT DEFAULT 'pending',
    sent_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    responded_at TIMESTAMP,
    expires_at TIMESTAMP,
    token TEXT UNIQUE NOT NULL,
    UNIQUE(workspace_id, invited_email)
);

### Workspace Logs
CREATE TABLE workspace_logs (
    id SERIAL PRIMARY KEY,
    workspace_id INT NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
    user_id INT,
    action TEXT NOT NULL,
    title TEXT NOT NULL,
    description TEXT NOT NULL,
    metadata JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

## Meeting System

### Meetings
CREATE TABLE meetings (
    id SERIAL PRIMARY KEY,
    workspace_id INT NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
    title TEXT NOT NULL,
    description TEXT,
    meeting_link TEXT,
    platform TEXT,
    location TEXT,
    start_time TIMESTAMP NOT NULL,
    end_time TIMESTAMP NOT NULL,
    duration INT NOT NULL,
    meeting_type TEXT DEFAULT 'scheduled',
    status TEXT DEFAULT 'scheduled',
    is_recurring BOOLEAN DEFAULT FALSE,
    recurrence_rule TEXT,
    created_by_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    reminder_sent BOOLEAN DEFAULT FALSE,
    transcript_url TEXT,
    recording_url TEXT,
    agenda TEXT,
    meeting_source TEXT DEFAULT 'kairo',
    ai_insights_generated BOOLEAN DEFAULT FALSE,
    metadata JSONB
);

### Meeting Participants
CREATE TABLE meeting_participants (
    id SERIAL PRIMARY KEY,
    meeting_id INT NOT NULL REFERENCES meetings(id) ON DELETE CASCADE,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    role TEXT DEFAULT 'participant',
    status TEXT DEFAULT 'invited',
    joined_at TIMESTAMP,
    left_at TIMESTAMP,
    response_time TIMESTAMP,
    UNIQUE(meeting_id, user_id)
);

### Meeting Notes
CREATE TABLE meeting_notes (
    id SERIAL PRIMARY KEY,
    meeting_id INT NOT NULL REFERENCES meetings(id) ON DELETE CASCADE,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    content TEXT NOT NULL,
    type TEXT DEFAULT 'manual',
    timestamp INT DEFAULT 0,
    color TEXT DEFAULT '#3b82f6',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    author_name TEXT
);

### Meeting Files
CREATE TABLE meeting_files (
    id SERIAL PRIMARY KEY,
    meeting_id INT NOT NULL REFERENCES meetings(id) ON DELETE CASCADE,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    filename TEXT NOT NULL,
    filepath TEXT NOT NULL,
    file_size BIGINT NOT NULL,
    mime_type TEXT,
    file_type TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    uploader_name TEXT
);

### Action Items
CREATE TABLE action_items (
    id SERIAL PRIMARY KEY,
    meeting_id INT NOT NULL REFERENCES meetings(id) ON DELETE CASCADE,
    title TEXT NOT NULL,
    description TEXT,
    assignee TEXT,
    due_date TIMESTAMP,
    status TEXT DEFAULT 'pending',
    confirmed_at TIMESTAMP,
    rejected_at TIMESTAMP,
    confirmed_by INT REFERENCES users(id),
    rejected_by INT REFERENCES users(id),
    canonical_key TEXT NOT NULL,
    confidence FLOAT DEFAULT 0.0,
    source_chunk INT,
    first_seen_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_seen_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    raw_data JSONB,
    update_history JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

### Notifications
CREATE TABLE notifications (
    id SERIAL PRIMARY KEY,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    title TEXT NOT NULL,
    message TEXT NOT NULL,
    type TEXT NOT NULL,
    priority TEXT DEFAULT 'medium',
    workspace TEXT,
    is_read BOOLEAN DEFAULT FALSE,
    action_required BOOLEAN DEFAULT FALSE,
    related_id TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP
);

## AI & Embeddings Core

### AI Insights
CREATE TABLE ai_insights (
    id UUID PRIMARY KEY,
    meeting_id TEXT NOT NULL,
    insight_type TEXT NOT NULL,
    content TEXT NOT NULL,
    confidence_score FLOAT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

### User Voice Embeddings (pgvector)
CREATE TABLE user_voice_embeddings (
    id UUID PRIMARY KEY,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    embedding VECTOR(192),
    version INT DEFAULT 1,
    snr_score FLOAT,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP
);

### Speaker Identity Maps
CREATE TABLE speaker_identity_maps (
    id UUID PRIMARY KEY,
    meeting_id INT NOT NULL REFERENCES meetings(id) ON DELETE CASCADE,
    speaker_label TEXT NOT NULL,
    user_id INT REFERENCES users(id) ON DELETE SET NULL,
    confidence_score FLOAT NOT NULL,
    tier_resolved INT NOT NULL,
    metadata JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    UNIQUE(meeting_id, speaker_label)
);

### Meeting Embeddings (pgvector)
CREATE TABLE meeting_embeddings (
    id UUID PRIMARY KEY,
    meeting_id INT NOT NULL REFERENCES meetings(id) ON DELETE CASCADE,
    content_type TEXT NOT NULL,
    content_id TEXT,
    content TEXT NOT NULL,
    embedding VECTOR(384),
    metadata JSONB,
    chunk_index INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP
);

### Meeting Memory Contexts (pgvector)
CREATE TABLE meeting_memory_contexts (
    id UUID PRIMARY KEY,
    meeting_id INT UNIQUE NOT NULL REFERENCES meetings(id) ON DELETE CASCADE,
    summary_embedding VECTOR(384),
    key_topics TEXT[],
    key_decisions JSONB,
    key_action_items JSONB,
    participants TEXT[],
    meeting_context TEXT NOT NULL,
    embedding_count INT DEFAULT 0,
    last_processed_at TIMESTAMP NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

### Meeting Relationships
CREATE TABLE meeting_relationships (
    id UUID PRIMARY KEY,
    source_meeting_id INT NOT NULL REFERENCES meetings(id) ON DELETE CASCADE,
    target_meeting_id INT NOT NULL REFERENCES meetings(id) ON DELETE CASCADE,
    relationship_type TEXT NOT NULL,
    similarity_score FLOAT NOT NULL,
    shared_topics TEXT[],
    shared_participants TEXT[],
    metadata JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(source_meeting_id, target_meeting_id)
);

## Task & Workflow System

### Kanban Columns
CREATE TABLE kanban_columns (
    id SERIAL PRIMARY KEY,
    workspace_id INT NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    position INT NOT NULL,
    is_default BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    UNIQUE(workspace_id, name)
);

### Tasks
CREATE TABLE tasks (
    id SERIAL PRIMARY KEY,
    workspace_id INT NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
    column_id INT NOT NULL REFERENCES kanban_columns(id) ON DELETE CASCADE,
    action_item_id INT UNIQUE REFERENCES action_items(id) ON DELETE SET NULL,
    title TEXT NOT NULL,
    description TEXT,
    assignee TEXT,
    due_date TIMESTAMP,
    priority TEXT DEFAULT 'medium',
    position INT NOT NULL,
    metadata JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP
);

### Tags
CREATE TABLE tags (
    id SERIAL PRIMARY KEY,
    workspace_id INT NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    color TEXT DEFAULT '#3b82f6',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    UNIQUE(workspace_id, name)
);

### Task Tags
CREATE TABLE task_tags (
    id SERIAL PRIMARY KEY,
    task_id INT NOT NULL REFERENCES tasks(id) ON DELETE CASCADE,
    tag_id INT NOT NULL REFERENCES tags(id) ON DELETE CASCADE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(task_id, tag_id)
);

### Sent Reminders
CREATE TABLE sent_reminders (
    id SERIAL PRIMARY KEY,
    task_id INT NOT NULL REFERENCES tasks(id) ON DELETE CASCADE,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    reminder_interval INT NOT NULL,
    sent_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(task_id, user_id, reminder_interval)
);
